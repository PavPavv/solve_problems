function bSearch(arr, target) {
  const sortedArr = arr.sort((a, b) => a - b);
  let l = 0;
  let r = sortedArr.length - 1;
  let result = -1;

  while (l <= r) {

      return result = middle;
    } else if (guess > target) {
      r = middle - 1;

      l = middle + 1;

  }

}
